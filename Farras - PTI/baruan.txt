#include <iostream>
#include <string>
#include <iomanip>

using namespace std;

// Fungsi untuk Menghitung Gaji Karyawan
void menghitungGajiKaryawan() {
    string NIK, NamaKaryawan, AlamatKaryawan;
    int Golongan;
    double TotalJamKerja, UangHarian, UangLembur, TotalLembur = 0, UangTransport, Gapok, GajiTotal, Pajak, GajiBersih;

    cout << "\n--- Input Data Gaji Karyawan ---" << endl;
    cout << "NIK: "; cin >> NIK;
    cout << "Nama Karyawan: "; cin.ignore(); getline(cin, NamaKaryawan);
    cout << "Alamat: "; getline(cin, AlamatKaryawan);
    cout << "Golongan (1/2/3): "; cin >> Golongan;
    cout << "Total Jam Kerja: "; cin >> TotalJamKerja;

    // Ketentuan UangHarian, UangLembur, dan Gapok
    if (Golongan == 1) {
        UangHarian = 20000; UangLembur = 10000; Gapok = 3000000;
    } else if (Golongan == 2) {
        UangHarian = 15000; UangLembur = 7500; Gapok = 2500000;
    } else {
        UangHarian = 10000; UangLembur = 5000; Gapok = 2000000;
    }

    // Hitung TotalLembur (TL)
    if (TotalJamKerja > 192) {
        TotalLembur = (TotalJamKerja - 192) * UangLembur;
    }

    // Hitung UangTransport
    UangTransport = UangHarian * 24;

    // Hitung Total, Pajak, dan Bersih
    GajiTotal = Gapok + UangTransport + TotalLembur;
    Pajak = 0.025 * GajiTotal;
    GajiBersih = GajiTotal - Pajak;

    // Output
    cout << fixed << setprecision(0);
    cout << "\n--- Hasil Perhitungan Gaji ---" << endl;
    cout << "NIK: " << NIK << "\nNama: " << NamaKaryawan << "\nAlamat: " << AlamatKaryawan << endl;
    cout << "Golongan: " << Golongan << "\nGapok: " << Gapok << endl;
    cout << "Total Lembur: " << TotalLembur << "\nUang Transport: " << UangTransport << endl;
    cout << "Gaji Total: " << GajiTotal << "\nPajak (2.5%): " << Pajak << "\nGaji Bersih: " << GajiBersih << endl;
}

// Fungsi untuk Menghitung Komisi Sales
void menghitungKomisiSales() {
    string NamaSales;
    double JumlahPenjualan, KomisiPenjualan = 0;

    cout << "\n--- Input Data Komisi Sales ---" << endl;
    cout << "Nama Sales: "; cin.ignore(); getline(cin, NamaSales);
    cout << "Jumlah Penjualan: "; cin >> JumlahPenjualan;

    if (JumlahPenjualan < 500000) {
        KomisiPenjualan = 0;
    } else if (JumlahPenjualan >= 500000 && JumlahPenjualan <= 1000000) {
        KomisiPenjualan = 0.10 * JumlahPenjualan;
    } else if (JumlahPenjualan > 1000000 && JumlahPenjualan <= 1500000) {
        KomisiPenjualan = 0.125 * JumlahPenjualan;
    } else {
        KomisiPenjualan = 0.15 * JumlahPenjualan;
    }

    cout << fixed << setprecision(0);
    cout << "\n--- Hasil Perhitungan Komisi ---" << endl;
    cout << "Nama Sales: " << NamaSales << endl;
    cout << "Jumlah Penjualan: " << JumlahPenjualan << endl;
    cout << "Komisi Penjualan: " << KomisiPenjualan << endl;
}
